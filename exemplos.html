Cálculo de Fatorial

Veja como calcular o fatorial de um número em R:

# Função para calcular fatorial
fatorial <- function(n) {
  if(n == 0) {
    return(1)
  } else {
    return(n * fatorial(n - 1))
  }
}

# Exemplo de uso
numero <- 5
resultado <- fatorial(numero)
print(paste("Fatorial de", numero, "é", resultado))

Verificação de Número Primo

Confira como verificar se um número é primo em R:

# Função para verificar se um número é primo
eh_primo <- function(n) {
  if (n <= 1) {
    return(FALSE)
  }
  for (i in 2:sqrt(n)) {
    if (n %% i == 0) {
      return(FALSE)
    }
  }
  return(TRUE)
}

# Exemplo de uso
numero <- 7
resultado <- eh_primo(numero)
if (resultado) {
  print(paste(numero, "é um número primo"))
} else {
  print(paste(numero, "não é um número primo"))
}

Operações Básicas

Veja como realizar operações matemáticas básicas em R:

# Definindo os números
a <- 15
b <- 3

# Operações básicas
soma <- a + b
subtracao <- a - b
multiplicacao <- a * b
divisao <- a / b

# Exibindo resultados
print(paste("Soma:", soma))
print(paste("Subtração:", subtracao))
print(paste("Multiplicação:", multiplicacao))
print(paste("Divisão:", divisao))
