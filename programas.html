Classe e Objeto em Ruby
                              Class Pessoa
                              Attr_accessor :nome, :idade
                              Def initialize(nome, idade)
                                @nome = nome
                                @idade = idade
                              End
                              Def apresentar
                                Puts “Olá, meu nome é #{@nome} e eu tenho #{@idade} anos.”
                              End
                            End
                            Pessoa1 = Pessoa.new(‘Maria’, 30)
                            Pessoa1.apresentar
